class ListNode:
    """Node class for the linked list"""
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next
    
    def __str__(self):
        """String representation for easy printing"""
        return str(self.val)

class LinkedList:
    """Linked List class with reverse functionality"""
    def __init__(self):
        self.head = None
    
    def append(self, val):
        """Add a new node to the end of the list"""
        new_node = ListNode(val)
        if not self.head:
            self.head = new_node
            return
        
        current = self.head
        while current.next:
            current = current.next
        current.next = new_node
    
    def reverse(self):
        """Reverse the linked list iteratively"""
        prev = None
        current = self.head
        
        while current:
            next_temp = current.next  # Store next node
            current.next = prev       # Reverse the link
            prev = current           # Move prev forward
            current = next_temp      # Move current forward
        
        self.head = prev  # Update head to the new first node
    
    def reverse_recursive(self):
        """Reverse the linked list recursively"""
        def _reverse_recursive(node):
            # Base case: empty list or single node
            if not node or not node.next:
                return node
            
            # Recursively reverse the rest of the list
            new_head = _reverse_recursive(node.next)
            
            # Reverse the current connection
            node.next.next = node
            node.next = None
            
            return new_head
        
        self.head = _reverse_recursive(self.head)
    
    def display(self):
        """Display the linked list"""
        if not self.head:
            return "Empty list"
        
        result = []
        current = self.head
        while current:
            result.append(str(current.val))
            current = current.next
        return " -> ".join(result)
    
    def to_list(self):
        """Convert linked list to Python list for easy verification"""
        result = []
        current = self.head
        while current:
            result.append(current.val)
            current = current.next
        return result
